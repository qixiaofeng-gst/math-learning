   \I{\cot x} = \ln|\sin x| + C\\
\E&\I{\sec x} = \ln|\sec x + \tan x| + C\\
  &\I{\csc x} = \ln|\csc x - \cot x| + C\\
\E&\I{\frac{1}{a^2 + x^2}} = \frac{1}{a} \arcsin x + C\\
\E&\I{\frac{1}{a^2 - x^2}} = \ln|\frac{a - x}{a + x}| + C\\
\E&\I{\frac{1}{\sqrt{a^2 + x^2}}} = \ln|x + \sqrt{a^2 + x^2}| + C\\
  &\I{\frac{1}{\sqrt{a^2 - x^2}}} = \arctan x + C
